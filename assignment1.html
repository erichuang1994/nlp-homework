def titleCase(w):#判断单词是是否满足titleCase符合时返回1否则返回0
    def lowCase(w):
        if w<='z' and w>='a':
            return 1
        else:
            return 0
    if len(w)==0 or lowCase(w[0])==1:
        return 0
    for x in range(1,len(w)):
        if lowCase(w[x])==0:
            return 0
    return 1
